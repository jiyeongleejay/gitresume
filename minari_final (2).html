def rev_to_word(raw_review):
    review_text = bs(raw_review, 'html.parser').get_text()
    letters_only = re.sub('[^a-zA-Z]', ' ', review_text)
    words = letters_only.lower().split()
    from nltk.corpus import stopwords
    stops = set(stopwords.words('english'))
    meaningful_words = [w for w in words if not w in stops]    
    meaningful_words = [w for w in meaningful_words if not w=='permalink']
    meaningful_words = [w for w in meaningful_words if not w=='help']
    meaningful_words = [w for w in meaningful_words if not w=='review']
    meaningful_words = [w for w in meaningful_words if not w=='found']
    stemming_words = [stemmer.stem(w) for w in meaningful_words]
    return( ' '.join(stemming_words))

def get_title_keywords(text):    
    min_count = 1   #     (  )
    max_length = 10 #   
    wordrank_extractor = KRWordRank(min_count=min_count, max_length=max_length)
    beta = 0.85    # PageRank decaying factor beta
    max_iter = 10
    keyword_list = []
    try:
        keywords, rank, graph = wordrank_extractor.extract([text], beta, max_iter)
        for word, r in sorted(keywords.items(), key=lambda x:x[1], reverse=True)[:30]:
                keyword_list.append(word)
#        keyword_list = []
#        for key, value in keywords.items():
#            if value >= 1:
#                keyword_list.append(key)
#        return keyword_list
    except ValueError:
        pass
    
    return keyword_list
